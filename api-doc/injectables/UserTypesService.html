import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { CreateUserTypeDto } from './dto/create-user-type.dto';
import { UpdateUserTypeDto } from './dto/update-user-type.dto';
import { UserTypeEntity } from './entities/user-type.entity';

@Injectable()
export class UserTypesService {
  constructor (
    @InjectRepository(UserTypeEntity) private readonly userTypeRepository: Repository<UserTypeEntity>,
  ){}
  async create(createUserTypeDto: CreateUserTypeDto) {
    const { name } = createUserTypeDto;
    const newData = { "name": name };
    const createNewUserType =  await this.userTypeRepository.save(newData)
    if( createNewUserType ){
      return {
        "data": createNewUserType,
        "status": 201,
        "message": "New Entry Created",
      };
    }else{
      return {status:"204", message:"No Data Created!"};
    }
  }

  async findAll() {
    const expectedData = await this.userTypeRepository.find();
    if( expectedData ){
      return {
        "data": expectedData,
        "status": 200,
        "message": "All Data Fetched",
      };
    }else{
      return {status:"204", message:"No Data Found!"};
    }
  }

  async findAllList() {
    const expectedData = await this.userTypeRepository.find({ 'status':1 });
    if( expectedData ){
      return {
        "data": expectedData,
        "status": 200,
        "message": "All Active Data Fetched",
      };
    }else{
      return {status:"204", message:"No Data Found!"};
    }
  }

  async findOne(id: string) {
    const expectedData = await this.userTypeRepository.findOne(id);
    if( expectedData ){
      return expectedData;
    }else{
      return {status:"204", message:"Data Deleted or Not Found!"};
    }
  }

  async update(id: string, updateUserTypeDto: UpdateUserTypeDto) {
    const expectedData = await this.userTypeRepository.findOne(id);
    if( expectedData ){
      await this.userTypeRepository.update(id, updateUserTypeDto);
      return await this.userTypeRepository.findOne(id);
    }else{
      return {status:"204", message:"Data Deleted or Not Found!"};
    }
  }

  async remove(id: string) {
    const expectedData = await this.userTypeRepository.findOne(id);
    if( expectedData ){
      const deletedData = await this.userTypeRepository.softDelete(id);
      if(deletedData.raw.changedRows == 1){
        return {status:"200", message:"Data has been Deleted!"};
      }else{
        return {status:"204", message:"Data can not Delete!"};
      }
    }else{
      return {status:"204", message:"Data Already Deleted or Not Found!"};
    }
  }

  async finalDelete(id: string) {
    const expectedData = await this.userTypeRepository.findOne(id);
    if( expectedData ){
      const deletedData = await this.userTypeRepository.delete(id);
      if(deletedData.raw.affectedRows > 0){
        return {status:"200", message:"Data has been Deleted!"};
      }else{
        return {status:"204", message:"Data can not Delete!"};
      }
    }else{
      return {status:"204", message:"Data Already Deleted or Not Found!"};
    }
  }
}
